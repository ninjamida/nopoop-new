This G-Code is optional. You can simply use the standard zMod start G-Code and leave change filament blank if you want.
However, do not forget to turn "Purge to prime tower" on!

This code is intended for use with USE_TRASH_ON_PRINT=0. However, it also works with USE_TRASH_ON_PRINT=2.

--== Start G-Code ==--
_VALIDATE_PRINT_SETTINGS CHECK_SCREEN=0 CHECK_TRASH=-2
START_PRINT EXTRUDER_TEMP=[nozzle_temperature_initial_layer] BED_TEMP=[bed_temperature_initial_layer_single]
SET_PRINT_STATS_INFO TOTAL_LAYER=[total_layer_count]



--== Filament Change G-Code ==--
{if toolchange_count > 1}
M104 S[old_filament_temp]
SET_VELOCITY_LIMIT ACCEL=[travel_acceleration]
_COLOR_CHANGE_PARAMS NOPOOP_MODE=1 NEW_MATERIAL_TEMPERATURE=[new_filament_temp]
T[next_extruder]
M109 S[new_filament_temp]
{if layer_num == 0}
SET_VELOCITY_LIMIT ACCEL=[initial_layer_acceleration]
{else}
SET_VELOCITY_LIMIT ACCEL=[default_acceleration]
{endif}
{else}
T[next_extruder]
{endif}
